114void calcularEnvolvente(float *senialFiltrada, float *envolvente, uint16_t tamanio)
115{
116 float alpha = 0.9; // Factor de suavizado
117 envolvente[0] = fabs(senialFiltrada[0]); // Inicializar la envolvente
118
119 for (uint16_t i = 1; i < tamanio; ++i)
120 {
121 envolvente[i] = alpha * fabs(senialFiltrada[i]) + (1 - alpha) * envolvente[i - 1]; // Suavizado exponencial
122 }
123}
