325void signal_Task(void *pvParameter)
326{
327 uint16_t dato; //12 bits
328 while (1)
329 {
330 ulTaskNotifyTake(pdTRUE, portMAX_DELAY); // Recibe una notificacion
331 AnalogInputReadSingle(CH1, &dato); // Se lee el dato del canal 1 y se guarda en "valor"
332 UartSendString(UART_PC, (char *)UartItoa(dato, 10)); // Se muestra el dato trasmitiendolo via UART al puerto serie (ver SerialOscilloscope)
333 UartSendString(UART_PC, "\r\n");
334 }
335}

347void reproducir_ECG_Task(void *pvParameter)

381 .port = UART_PC,
382 .baud_rate = 115200,
383 .func_p = NULL,
384 .param_p = NULL};
385 TimerInit(&timer_medicion_senial);
386 TimerInit(&timer_reproducir_ECG);
387

389 AnalogOutputInit();

393 xTaskCreate(&reproducir_ECG_Task, "senial_ecg", 2048, NULL, 5, &ECG_task_handle);
