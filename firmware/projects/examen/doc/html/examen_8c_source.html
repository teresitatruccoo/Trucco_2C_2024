287void medicion_caida_Task(void *pvParameter)
288{
289uint16_t aceleraciontotal=0;
290float umbral_caida=(4*0.3)+1.65; //4G umbral convertido por la sensibilidad.
291
292 while (true)
293 {
294 ulTaskNotifyTake(pdTRUE, portMAX_DELAY); /* La tarea espera en este punto hasta recibir una notificación */
295 {
296 AnalogInputReadSingle(CH1, &signalACELEROMETRO_X);
297 AnalogInputReadSingle(CH2, &signalACELEROMETRO_Y);
298 AnalogInputReadSingle(CH3, &signalACELEROMETRO_Z);
299 signalACELEROMETRO_X=abs(signalACELEROMETRO_X)+1.65;
300 signalACELEROMETRO_Y=abs(signalACELEROMETRO_Y)+1.65;
301 signalACELEROMETRO_Z=abs(signalACELEROMETRO_Z)+1.65;
302 aceleraciontotal=signalACELEROMETRO_X+signalACELEROMETRO_Y+signalACELEROMETRO_Z;
303 if (aceleraciontotal > umbral_caida)
304 {
305 UartSendString(UART_CONNECTOR, "Caida detectada \r\n");
306 }
307 }
308
309 }
