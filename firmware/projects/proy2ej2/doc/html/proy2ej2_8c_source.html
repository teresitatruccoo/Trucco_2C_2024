136static void mostrar(void *pvParameter)
137{
138

185void app_main(void)
186{
187 // Inicializo LCD, Switchs y LEDS
188 LcdItsE0803Init();
189 LedsInit();
190 HcSr04Init(GPIO_3, GPIO_2);
191 SwitchesInit();
192
193 timer_config_t timer_medicion = {
194 .timer = TIMER_A,
195 .period = CONFIG_BLINK_PERIOD_MEDICION_US, // De este tiempo a 0.
196 .func_p = FuncTimerA,
197 .param_p = NULL};
198
199 TimerInit(&timer_medicion);
200
201 SwitchActivInt(SWITCH_1, &InterrupcionTecla1, NULL);
202 SwitchActivInt(SWITCH_2, &InterrupcionTecla2, NULL);
203 // Se crean las tareas
204 xTaskCreate(&mostrar, "mostrar", 512, NULL, 5, &mostrar_task_handle);
205 xTaskCreate(&medicion, "medicion", 512, NULL, 5, &medir_task_handle);
206
207 TimerStart(timer_medicion.timer);
208}
